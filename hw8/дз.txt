c)
function a(x) {
    return x * 2;
}
var a;
alert(a);
При объявлении "а" в неё попадает ссылка на функцию поэтому при вызове "а" выводится содержимое функции, как обычный текст.


d)

    arguments[2] = 10;
    alert(a);
}
b(1, 2, 3);
В функцию аргументы передаются как массив значений, поэтому им можно манипулировать, как  элементами массива. В данном случае заменяем третий аргумент "а" с индексом 2 на 10.

надо написать bindFunc

function bindFunc(func, a){
   return func.bind(null, a);
}
var add = function(a, b) { return a + b; }
var add2 = bindFunc(add, 2);

add2(6) // outputs 8
add2(10) // outputs 12
